local UI = Lib:Create{
   Theme = "Dark", -- or any other theme
   Size = UDim2.new(0, 555, 0, 400) -- default
}

local Main = UI:Tab{
   Name = "Início"
}

local Divider = Main:Divider{
   Name = "Início shit"
}

local QuitDivider = Main:Divider{
   Name = "Sair"
}


local function UpdateESP(player)
    local ESP = ESPObjects[player]
    if ESP then
        local Vector, OnScreen = Camera:WorldToViewportPoint(player.Character.Head.Position)
        if OnScreen then
            ESP.Position = Vector2.new(Vector.X, Vector.Y - 25)
            ESP.Text = player.Name
            ESP.Visible = true
        else
            ESP.Visible = false
        end
    end
end

local function ToggleESP()
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= Players.LocalPlayer then
            UpdateESP(player)
        end
    end
end

local function SetESPColor(player)
    local color
    if player.TeamColor == Players.LocalPlayer.TeamColor then
        color = Color3.new(0, 0, 1) -- Azul para aliados
    else
        color = player.TeamColor == BrickColor.new("Bright red") and Color3.new(1, 0, 0) or Color3.new(0, 0, 0) -- Vermelho para inimigos, preto para outros
    end
    return color
end

local function OnKeyPress(input)
    if input.KeyCode == Enum.KeyCode.E then
        for _, ESP in pairs(ESPObjects) do
            ESP.Visible = not ESP.Visible
        end
    end
end

for _, player in pairs(Players:GetPlayers()) do
    if player ~= Players.LocalPlayer then
        CreateESP(player, SetESPColor(player))
    end
end

UserInputService.InputBegan:Connect(OnKeyPress)
RunService.RenderStepped:Connect(ToggleESP)
